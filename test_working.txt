/* If/else */
var d int = 5;
if (d > 3) {
    print(d);
} else {
    INC d;
    print(d);
}

/* else-if */
var score int = 85;
if (score > 90) {
    print(1);
} else if (score > 80) {
    print(2);
} else {
    print(3);
}
